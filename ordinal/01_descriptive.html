Show histograms#

cols = list(data)
fig = plt.figure(figsize=(12, 12))

for n, col in enumerate(cols):
    ax = plt.subplot(3, 4, n + 1)
    ax.hist(data[col])
    ax.set_xlabel(col)
    ax.set_ylabel('frequency')

plt.tight_layout(pad=2)
plt.show()
